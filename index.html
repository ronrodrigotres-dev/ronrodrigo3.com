async function ejecutarAuditoria() { const email = document.getElementById('email').value; const url = document.getElementById('url').value; const btn = document.querySelector('button[onclick]'); const scoreNum = document.getElementById('score-number'); const gauge = document.querySelector('.score-gauge'); const btnPdf = document.getElementById('btn-pdf'); if (!email || !url) { return alert("CRÍTICO: Email y URL requeridos."); } btn.innerText = "PROCESANDO LÓGICA..."; btn.disabled = true; try { const response = await fetch(`${API_BASE_URL}/analizar-sitio`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ email: email, url: url }) }); const data = await response.json(); if (data.status === "success") { currentAuditId = data.audit_id; gauge.style.setProperty('--percentage', `${data.health_score.score}%`); scoreNum.innerText = `${data.health_score.score}/100`; btnPdf.disabled = false; btnPdf.classList.remove('opacity-20', 'cursor-not-allowed'); } } catch (e) { alert("ERROR: Motor autónomo inaccesible."); } btn.innerText = "INICIAR AUDITORÍA"; btn.disabled = false; }

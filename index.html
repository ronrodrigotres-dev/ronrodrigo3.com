async function ejecutarAuditoria() { const email = document.getElementById('email').value; const url = document.getElementById('url').value; const btn = document.getElementById('btn-auditoria'); const btnText = document.getElementById('btn-text'); const btnFill = document.getElementById('btn-fill'); const scoreNum = document.getElementById('score-number'); const gauge = document.querySelector('.score-gauge'); const btnPdf = document.getElementById('btn-pdf'); if (!email || !url) { return alert("CRÍTICO: Email y URL requeridos."); } btn.disabled = true; btn.classList.add("processing"); btnText.innerText = "PROCESANDO..."; let progreso = 0; const intervalo = setInterval(() => { if (progreso < 95) { progreso += 5; btnFill.style.width = progreso + "%"; } }, 120); try { const response = await fetch(`${API_BASE_URL}/analizar-sitio`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ email: email, url: url }) }); const data = await response.json(); clearInterval(intervalo); btnFill.style.width = "100%"; if (data.status === "success") { currentAuditId = data.audit_id; gauge.style.setProperty('--percentage', `${data.health_score.score}%`); scoreNum.innerText = `${data.health_score.score}/100`; btnPdf.disabled = false; btnPdf.classList.remove('opacity-20', 'cursor-not-allowed'); btnText.innerText = "ANÁLISIS COMPLETADO"; } } catch (e) { clearInterval(intervalo); btnFill.style.width = "0%"; btnText.innerText = "ERROR"; alert("ERROR: Motor autónomo inaccesible."); } setTimeout(() => { btnFill.style.width = "0%"; btnText.innerText = "INICIAR AUDITORÍA"; btn.classList.remove("processing"); btn.disabled = false; }, 2000); }
